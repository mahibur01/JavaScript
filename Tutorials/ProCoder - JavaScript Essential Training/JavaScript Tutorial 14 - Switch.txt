JavaScript Tutorial 14 - Switch

1. Case is used for small fixed data conditionaly switch 
Example: 

  let day = 0;

        switch(day){
            case 0:
                day = "Sunday";
            break;
            
            case 1:
                day = "Monday";
            break;
            
            case 2:
                day = "Tuesday";
            break;
            
            case 3:
                day = "Wednessday";
            break;
            
            case 4:
                day = "Thursday";
            break;
            
            case 5:
                day = "Friday";
            break;
            
            case 6:
                day = "Saturday";
            
        }

        console.log(day);
